`nxenv-symbols` provides access to the [SF Symbols](https://developer.apple.com/sf-symbols/) library on iOS.

## Usage

```jsx App.js
import { SymbolView, SymbolViewProps, SFSymbol } from 'nxenv-symbols';
import { StyleSheet, View } from 'react-native';

nxenvrt default function App() {
  return (
    <View style={styles.container}>
      <SymbolView name="airpods.chargingcase" style={styles.symbol} type="hierarchical" />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'center',
  },
  symbol: {
    width: 35,
    height: 35,
    margin: 5,
  },
});
```
